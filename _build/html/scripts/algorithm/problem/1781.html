import heapq

n = int(input())
tasks = []
for _ in range(n):
    dead, cup = map(int, input().split())
    tasks.append((dead, cup))
tasks.sort()
heap = []
for d, c in tasks:
    heapq.heappush(heap, c)
    if len(heap) > d:
        heapq.heappop(heap)
print(sum(heap))
