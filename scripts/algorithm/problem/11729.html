import sys, os

problem_num = "11729"

path = os.getcwd() + f"\\txt\\{problem_num}" + ".txt"
sys.stdin = open(path, "r")
input = sys.stdin.readline

n = int(input())

result = []


def move(start, to):
    return [start, to]


def hanoi(n, start, to, via):
    if n == 1:
        result.append(move(start, to))
    else:
        hanoi(n - 1, start, via, to)
        result.append(move(start, to))
        hanoi(n - 1, via, to, start)
    return result

answer = hanoi(n, 1, 3, 2)
print(len(answer))
for i in answer:
    print(*i)
