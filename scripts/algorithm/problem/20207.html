import sys, os

# sol
n = int(input())



cal = [0] * 365


for _ in range(n):
    s, e = map(int, input().split())
    for i in range(s - 1, e):
        cal[i] += 1



result = 0
i = 0



while i < 365:
    if cal[i] > 0:
        width = 0
        height = 0


        while i < 365 and cal[i] > 0:
            width += 1
            height = max(height, cal[i])
            i += 1
        result += width * height
    else:

        i += 1



print(result)
