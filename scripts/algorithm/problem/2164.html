from collections import deque
import os, sys


def step1(dq):
    dq.popleft()
    return dq

def step2(dq):
    dq.rotate(-1)
    return dq

problem_num = 2164
path = os.getcwd() + f'\\txt\\{problem_num}'  + '.txt'
sys.stdin = open(path,'r')
input = sys.stdin.readline()
n = int(input)
n_dq = deque([i for i in range(1,n+1)])
while len(n_dq) != 1:
    n_dq = step1(n_dq)
    n_dq = step2(n_dq)
print(*n_dq)

if n_dq[0] == 4:
    print('right')
else:
    print('wrong')
